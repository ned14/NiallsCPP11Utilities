55 inline unsigned _mm_movemask_epi8_neon(uint32x4_t Input)
56 {
57  static const uint8_t __attribute__ ((aligned (16))) _Powers[16]=
58  { 1, 2, 4, 8, 16, 32, 64, 128, 1, 2, 4, 8, 16, 32, 64, 128 };
59 
60  // Set the powers of 2 (do it once for all, if applicable)
61  static uint8x16_t Powers= vld1q_u8(_Powers);
62 
63  // Compute the mask from the input
64  uint64x2_t Mask= vpaddlq_u32(vpaddlq_u16(vpaddlq_u8(vandq_u8(vreinterpretq_u8_u32(Input), Powers))));
65 
66  // Get the resulting bytes
67  uint16_t Output;
68  vst1q_lane_u8((uint8_t*)&Output + 0, (uint8x16_t)Mask, 0);
69  vst1q_lane_u8((uint8_t*)&Output + 1, (uint8x16_t)Mask, 8);
70 
71  return (unsigned) Output;
72 }
