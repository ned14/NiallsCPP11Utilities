128 template<typename T> struct has_call_operator
129 {
130  template<typename F, typename A> // detect 1-arg operator()
131  static char test(int, decltype( (*(F*)0)( (*(A*)0) ) ) = 0);
132 
133  template<typename F, typename A, typename B> // detect 2-arg operator()
134  static char test(int, decltype( (*(F*)0)( (*(A*)0), (*(B*)0) ) ) = 0);
135 
136  template<typename F, typename A, typename B, typename C> // detect 3-arg operator()
137  static char test(int, decltype( (*(F*)0)( (*(A*)0), (*(B*)0), (*(C*)0) ) ) = 0);
138 
139  template<typename F, typename ...Args> // worst match
140  static char (&test(...))[2];
141 
142  static constexpr bool OneArg = (sizeof( test<T, int>(0) ) == 1);
143  static constexpr bool TwoArg = (sizeof( test<T, int, int>(0) ) == 1);
144  static constexpr bool ThreeArg = (sizeof( test<T, int, int, int>(0) ) == 1);
145 
146  static constexpr bool HasTemplatedOperator = OneArg || TwoArg || ThreeArg;
147  static constexpr bool value = Impl::has_regular_call_operator<HasTemplatedOperator, T>::value;

200  template<class _registry, class _type, class _containertype> struct StaticTypeRegistryStorage
201  {
202  typedef _registry registry;
203  typedef _type type;
204  typedef _containertype containertype;
205  static containertype **registryStorage()
206  {
207  static containertype **_registryStorage; // Keep a local cache
208  if(!_registryStorage)
209  {
210  const std::type_info &typeinfo=typeid(containertype);
211  // This deliberately and intentionally leaks because we have no way of knowing when to clean it up
212  if(!static_type_registry_storage)
213  static_type_registry_storage=new ErasedTypeRegistryMapType;
214  auto typemap=(*static_type_registry_storage)[typeinfo.hash_code()];
215  auto containerstorage=typemap[typeinfo.name()];
216  if(!containerstorage)
217  containerstorage=static_cast<void *>(new containertype());
218  _registryStorage=(containertype **) &containerstorage;
219  }
220  return _registryStorage;
221  }
222  static void RegisterData(type &&c)
223  {
224  (*registryStorage())->push_back(std::move(c));
225  }
226  static void UnregisterData(type &&c)
227  {
228  auto _r=registryStorage();
229  auto r=*_r;
230  // Quick optimisation for tail pop to avoid a search
231  if(*r->rbegin()==c)
232  r->erase(--r->end());
233  else
234  r->erase(std::remove(r->begin(), r->end(), std::forward<type>(c)), r->end());
235  if(r->empty())

238  *_r=nullptr;
