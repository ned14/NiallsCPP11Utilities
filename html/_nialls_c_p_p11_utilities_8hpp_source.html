144  template<class _registry, class _type, class _containertype> struct StaticTypeRegistryStorage
145  {
146  typedef _registry registry;
147  typedef _type type;
148  typedef _containertype containertype;
149  static containertype **registryStorage()
150  {
151  static containertype **_registryStorage; // Keep a local cache
152  if(!_registryStorage)
153  {
154  const std::type_info &typeinfo=typeid(containertype);
155  // This deliberately and intentionally leaks because we have no way of knowing when to clean it up
156  if(!static_type_registry_storage)
157  static_type_registry_storage=new ErasedTypeRegistryMapType;
158  auto typemap=(*static_type_registry_storage)[typeinfo.hash_code()];
159  auto containerstorage=typemap[typeinfo.name()];
160  if(!containerstorage)
161  containerstorage=static_cast<void *>(new containertype());
162  _registryStorage=(containertype **) &containerstorage;
163  }
164  return _registryStorage;
165  }
166  static void RegisterData(type &&c)
167  {
168  (*registryStorage())->push_back(std::move(c));
169  }
170  static void UnregisterData(type &&c)
171  {
172  auto _r=registryStorage();
173  auto r=*_r;
174  // Quick optimisation for tail pop to avoid a search
175  if(*r->rbegin()==c)
176  r->erase(--r->end());
177  else
178  r->erase(std::remove(r->begin(), r->end(), std::forward<type>(c)), r->end());
179  if(r->empty())
180  {
181  delete r;
182  *_r=nullptr;
183  }
184  }
185  };
