153 template<typename T> struct has_call_operator
154 {
155  template<typename F, typename A> // detect 1-arg operator()
156  static char test(int, decltype( (*(F*)0)( (*(A*)0) ) ) = 0);
157 
158  template<typename F, typename A, typename B> // detect 2-arg operator()
159  static char test(int, decltype( (*(F*)0)( (*(A*)0), (*(B*)0) ) ) = 0);
160 
161  template<typename F, typename A, typename B, typename C> // detect 3-arg operator()
162  static char test(int, decltype( (*(F*)0)( (*(A*)0), (*(B*)0), (*(C*)0) ) ) = 0);
163 
164  template<typename F, typename ...Args> // worst match
165  static char (&test(...))[2];
166 
167  static constexpr bool OneArg = (sizeof( test<T, int>(0) ) == 1);
168  static constexpr bool TwoArg = (sizeof( test<T, int, int>(0) ) == 1);
169  static constexpr bool ThreeArg = (sizeof( test<T, int, int, int>(0) ) == 1);
170 
171  static constexpr bool HasTemplatedOperator = OneArg || TwoArg || ThreeArg;
172  static constexpr bool value = Impl::has_regular_call_operator<HasTemplatedOperator, T>::value;
