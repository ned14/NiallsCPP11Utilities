119  template<class _registry, class _type, class _containertype> struct StaticTypeRegistryStorage
120  {
121  typedef _registry registry;
122  typedef _type type;
123  typedef _containertype containertype;
124  static containertype **registryStorage()
125  {
126  static containertype **_registryStorage; // Keep a local cache
127  if(!_registryStorage)
128  {
129  const type_info &typeinfo=typeid(containertype);
130  // This deliberately and intentionally leaks because we have no way of knowing when to clean it up
131  if(!static_type_registry_storage)
132  static_type_registry_storage=new ErasedTypeRegistryMapType;
133  auto typemap=(*static_type_registry_storage)[typeinfo.hash_code()];
134  auto containerstorage=typemap[typeinfo.name()];
135  if(!containerstorage)
136  containerstorage=static_cast<void *>(new containertype());
137  _registryStorage=(containertype **) &containerstorage;
138  }
139  return _registryStorage;
140  }
141  static void RegisterData(type &&c)
142  {
143  (*registryStorage())->push_back(std::move(c));
144  }
145  static void UnregisterData(type &&c)
146  {
147  auto _r=registryStorage();
148  auto r=*_r;
149  // Quick optimisation for tail pop to avoid a search
150  if(*r->rbegin()==c)
151  r->erase(--r->end());
152  else
153  r->erase(std::remove(r->begin(), r->end(), std::forward<type>(c)), r->end());
154  if(r->empty())
155  {
156  delete r;
157  *_r=nullptr;
158  }
159  }
